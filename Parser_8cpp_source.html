40Expr Parser::assignment() {
41 Expr expr = logicOr();
42
43 if (match(TokenType::EQUAL)) {

85Expr Parser::equality() {
86 Expr expr = comparison();
87
88 while (match(TokenType::BANG_EQUAL, TokenType::EQUAL_EQUAL)) {
89 Token op = previous();
90 if (isExprBinary(expr)) {
91 auto binary = std::get<Box<Binary>>(expr);
92 if (binary->op.type == TokenType::BANG_EQUAL ||
93 binary->op.type == TokenType::EQUAL_EQUAL) {
94 error(op, std::format("Chained equality of the form 'a %s b %s c' is "
95 "illegal. Use 'a %s b and b %s c' instead.",
96 binary->op.lexeme, op.lexeme, binary->op.lexeme,
97 op.lexeme));
98 }
99 }
100
101 Expr right = comparison();
102 expr = Binary{expr, op, right};
103 }
104
105 return expr;
106}

109Expr Parser::comparison() {
110 Expr expr = term();
111
112 while (match(TokenType::GREATER, TokenType::GREATER_EQUAL, TokenType::LESS,
113 TokenType::LESS_EQUAL)) {
114 Token op = previous();
115 if (isExprBinary(expr)) {
116 auto binary = std::get<Box<Binary>>(expr);
117 if (binary->op.type == TokenType::GREATER ||
118 binary->op.type == TokenType::GREATER_EQUAL ||
119 binary->op.type == TokenType::LESS ||
120 binary->op.type == TokenType::LESS_EQUAL) {
121 error(op, std::format("Chained comparison of the form 'a %s b %s c' is "
122 "illegal. Use 'a %s b and b %s c' instead.",
123 binary->op.lexeme, op.lexeme, binary->op.lexeme,
124 op.lexeme));
125 }
126 }
127
128 Expr right = term();
129 expr = Binary{expr, op, right};
130 }
131
132 return expr;
133}

199Expr Parser::finishCall(const Expr &callee) {
200 ListOfExprs arguments{};
201
202 if (!check(TokenType::RIGHT_PAREN)) {
203 do {
204 arguments.emplace_back(expression());
205 } while (match(TokenType::COMMA));
206 }
207
208 Token paren = consume(TokenType::RIGHT_PAREN, "Expect ')' after arguments.");
209
210 return Call{callee, paren, arguments};
211}

219Expr Parser::primary() {
220 // boring keywords
221 if (match(TokenType::FALSE)) {
222 return Literal{false};
223 }
224 if (match(TokenType::TRUE)) {
225 return Literal{true};
226 }
227
228 // actual literal types
229 if (match(TokenType::INTEGER, TokenType::FLOAT))
230 return Literal{previous().literal, previous()};
231
232 // ( expr )
233 if (match(TokenType::LEFT_PAREN)) {
234 Token paren = previous();
235 Expr expr = expression();
236 consume(TokenType::RIGHT_PAREN, "Expect ')' after expression.");
237 return Grouping{expr, paren};
238 }
239
240 if (match(TokenType::IDENTIFIER)) {
241 return Variable(previous());
242 }
243
244 throw error(peek(), "Expect expression.");
245}
