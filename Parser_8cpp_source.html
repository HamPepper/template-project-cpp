86Expr Parser::equality() {
87 Expr expr = comparison();
88
89 while (match(TokenType::BANG_EQUAL, TokenType::EQUAL_EQUAL)) {
90 Token op = previous();
91 if (isExprBinary(expr)) {
92 auto binary = std::get<Box<Binary>>(expr);
93 if (binary->op.type == TokenType::BANG_EQUAL ||
94 binary->op.type == TokenType::EQUAL_EQUAL) {
95 error(op, std::format("Chained equality of the form 'a {} b {} c' is "
96 "illegal. Use 'a {} b & b {} c' instead.",
97 binary->op.lexeme, op.lexeme, binary->op.lexeme,
98 op.lexeme));
99 }
100 }
101
102 Expr right = comparison();
103 expr = Binary{expr, op, right};
104 }
105
106 return expr;
107}

110Expr Parser::comparison() {
111 Expr expr = term();
112
113 while (match(TokenType::GREATER, TokenType::GREATER_EQUAL, TokenType::LESS,
114 TokenType::LESS_EQUAL)) {
115 Token op = previous();
116 if (isExprBinary(expr)) {
117 auto binary = std::get<Box<Binary>>(expr);
118 if (binary->op.type == TokenType::GREATER ||
119 binary->op.type == TokenType::GREATER_EQUAL ||
120 binary->op.type == TokenType::LESS ||
121 binary->op.type == TokenType::LESS_EQUAL) {
122 error(op, std::format("Chained comparison of the form 'a {} b {} c' is "
123 "illegal. Use 'a {} b & b {} c' instead.",
124 binary->op.lexeme, op.lexeme, binary->op.lexeme,
125 op.lexeme));
126 }
127 }
128
129 Expr right = term();
130 expr = Binary{expr, op, right};
131 }
132
133 return expr;
134}

200Expr Parser::finishCall(const Expr &callee) {
201 ListOfExprs arguments{};
202
203 if (!check(TokenType::RIGHT_PAREN)) {
204 do {
205 arguments.emplace_back(expression());
206 } while (match(TokenType::COMMA));
207 }
208
209 Token paren = consume(TokenType::RIGHT_PAREN, "Expect ')' after arguments.");
210
211 return Call{callee, paren, arguments};

220Expr Parser::primary() {
221 // boring keywords
222 if (match(TokenType::FALSE))
223 return Literal{false};
224 if (match(TokenType::TRUE))
225 return Literal{true};
226
227 // actual literal types
228 if (match(TokenType::INTEGER, TokenType::FLOAT))
229 return Literal{previous().literal, previous()};
230
231 // ( expr )
232 if (match(TokenType::LEFT_PAREN)) {
233 Token paren = previous();
234 Expr expr = expression();
235 consume(TokenType::RIGHT_PAREN, "Expect ')' after expression.");
236 return Grouping{expr, paren};
237 }
238
239 if (match(TokenType::IDENTIFIER)) {
240 return Variable(previous());
241 }
242
243 throw error(peek(), "Expect expression.");
244}
