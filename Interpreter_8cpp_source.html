57Object Interpreter::evaluate(const std::string &source) {
58 Scanner scanner(*this, source);
59 auto tokens = scanner.scanTokens();
60
61 Parser parser(*this, tokens);
62 auto exprs = parser.parse();
63
64#ifdef DEBUG_PARSER
65 PrinterExpr printer{};
66 for (const auto &e : exprs)
67 printer.print(e);
68#endif
69
70 Object result{};
71 EvaluatorExpr evaluator{*this};
72 try {
73 for (const auto &e : exprs)
74 result = evaluator(e);
75 } catch (const RuntimeError &e) {
76 runtimeError(e);
77 }
78
79 return result;
80}
