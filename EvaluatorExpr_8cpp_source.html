23Object EvaluatorExpr::operator()(Box<Binary> expr) {
24 Object left = (*this)(expr->left);
25 Object right = (*this)(expr->right);
26
27 // do check first
28 switch (expr->op.type) {
29 case TokenType::PLUS:
30 case TokenType::MINUS:
31 case TokenType::STAR:
32 case TokenType::SLASH:
33 case TokenType::CARET:
34 case TokenType::GREATER:
35 case TokenType::GREATER_EQUAL:
36 case TokenType::LESS:
37 case TokenType::LESS_EQUAL:
38 if (!isNumber(left) && !isNumber(right))
39 throw RuntimeError{
40 std::format("Operands must be numbers for '{}'.", expr->op.lexeme)};
41 break;
42
43 default:
44 break;
45 }
46
47 // now, the actual operation
48 switch (expr->op.type) {
49 case TokenType::PLUS:
50 return left + right;
51 case TokenType::MINUS:
52 return left - right;
53 case TokenType::STAR:
54 return left * right;
55 case TokenType::SLASH:
56 return left / right;
57 case TokenType::CARET:
58 return power(left, right);
59
60 case TokenType::GREATER:
61 return left > right;
62 case TokenType::GREATER_EQUAL:
63 return left >= right;
64 case TokenType::LESS:
65 return left < right;
66 case TokenType::LESS_EQUAL:
67 return left <= right;
68 case TokenType::EQUAL_EQUAL:
69 return left == right;
70 case TokenType::BANG_EQUAL:
71 return left != right;
72
73 default:
74 throw RuntimeError{
75 std::format("Unknown binary operator '{}'.", expr->op.lexeme)};
76 }
77}

79Object EvaluatorExpr::operator()(Box<Call> expr) {
80 auto callee = toCallable((*this)(expr->callee));
81
82 std::vector<Object> arguments{};
83 for (auto &&arg : expr->arguments)
84 arguments.push_back((*this)(arg));
85
86 if (arguments.size() != callee->arity())
87 throw RuntimeError{std::format("Expected {} arguments but got {}.",
88 callee->arity(), arguments.size())};
89
90 return callee->call(arguments);

114Object EvaluatorExpr::operator()(Box<Unary> expr) {
115 Object right = (*this)(expr->right);
116
117 switch (expr->op.type) {
118 case TokenType::MINUS:
119 if (!isNumber(right))
120 throw RuntimeError{
121 std::format("Operand must be number for '{}'.", expr->op.lexeme)};
122 return -right;
123 case TokenType::BANG:
124 return Object{!right};

126 throw RuntimeError{
