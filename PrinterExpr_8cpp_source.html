14std::string PrinterExpr::operator()(Expr expr) {
15 auto hdlAssign = [this](Box<Assign> e) -> std::string {
16 return std::format("(assign {} {})", e->name.lexeme, operator()(e->value));
17 };
18 auto hdlBinary = [this](Box<Binary> e) -> std::string {
19 return std::format("({} {} {})",
20 e->op.lexeme, operator()(e->left), operator()(e->right));
21 };
22 auto hdlCall = [this](Box<Call> e) -> std::string {
23 std::string args;
24 for (auto it = e->arguments.begin(); it != e->arguments.end(); it++) {
25 args += operator()(*it) +
26 (std::next(it) == e->arguments.end() ? "" : " ");
27 }
28 return std::format("(call {} ({}))", operator()(e->callee), args);
29 };
30 auto hdlGrouping = [this](Box<Grouping> e) -> std::string {
31 return std::format("(group {})", operator()(e->expression));
32 };
33 auto hdlLiteral = [](Box<Literal> e) -> std::string {
34 return toString(e->value);
35 };
36 auto hdlLogical = [this](Box<Logical> e) -> std::string {
37 return std::format("({} {} {})",
38 e->op.lexeme, operator()(e->left), operator()(e->right));
39 };
40 auto hdlUnary = [this](Box<Unary> e) -> std::string {
41 return std::format("({} {})", e->op.lexeme, operator()(e->right));
42 };
43 auto hdlVariable = [](Box<Variable> e) -> std::string {
44 return e->name.lexeme;
45 };
46

48 hdlLiteral, hdlLogical, hdlUnary, hdlVariable},
